/**
 * @public
 */
export class Named<
  C extends Named.Color = Named.Color,
> extends Format<"named"> {
  public static of<C extends Named.Color>(color: C): Named<C> {
    return new Named(color);
  }
 
  private readonly _color: C;
 
  protected constructor(color: C) {
    super("named");
    this._color = color;
  }
 
  public get color(): C {
    return this._color;
  }
 
  public get value(): number {
    return Colors[this._color];
  }
 
  public get red(): Number.Fixed {
    return Number.of(this.value >>> 16);
  }
 
  public get green(): Number.Fixed {
    return Number.of((this.value >>> 8) & 0xff);
  }
 
  public get blue(): Number.Fixed {
    return Number.of(this.value & 0xff);
  }
 
  public get alpha(): Number.Fixed {
    // The "transparent" color has an alpha of 0 as it's, well, transparent. All
    // other named colors are fully opaque and therefore have an alpha of 1.
    return Number.of(this._color === "transparent" ? 0 : 1);
  }
 
  public resolve(): RGB.Canonical {
    return RGB.of(
      ...Format.resolve(this.red, this.green, this.blue, this.alpha),
    );
  }
 
  public equals(value: unknown): value is this {
    return value instanceof Named && value._color === this._color;
  }
 
  public hash(hash: Hash): void {
    hash.writeString(this._color);
  }
 
  public toJSON(): Named.JSON {
    return {
      ...super.toJSON(),
      color: this._color,
    };
  }
 
  public toString(): string {
    return this._color;
  }
}
